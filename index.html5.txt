<body>
  <h1>Vapi Chat Demo</h1>

  <div id="log"></div>

  <form id="form">
    <input id="input" placeholder="Nachricht…" autocomplete="off" />
    <button type="submit">Senden</button>
  </form>

  <script>
    // 1) HIER deinen Öffentlichen Schlüssel einfügen:
    const VAPI_PUBLIC_KEY = "PASTE_PUBLIC_KEY_HERE";

    // 2) HIER deine Assistant ID einfügen (z.B. "asst_...")
    const ASSISTANT_ID = "PASTE_ASSISTANT_ID_HERE";

    const log = document.getElementById("log");
    const form = document.getElementById("form");
    const input = document.getElementById("input");

    function addLine(who, text) {
      const div = document.createElement("div");
      div.className = "msg";
      div.innerHTML = `<span class="${who}">${who}:</span> ${text}`;
      log.appendChild(div);
      log.scrollTop = log.scrollHeight;
    }

    form.addEventListener("submit", async (e) => {
      e.preventDefault();
      const text = input.value.trim();
      if (!text) return;

      addLine("user", text);
      input.value = "";

      const res = await fetch("https://api.vapi.ai/chat", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Authorization": `Bearer ${VAPI_PUBLIC_KEY}`
        },
        body: JSON.stringify({
          assistantId: ASSISTANT_ID,
          input: text
        })
      });

      const data = await res.json();
      const out = data?.output?.[0]?.content ?? JSON.stringify(data);
      addLine("assistant", out);
    });
  </script>
